\nomenclature{$A$}{4-bit input operand A to the ALU}
\nomenclature{$B$}{4-bit input operand B to the ALU}
\nomenclature{$S_i$}{Control signal for operation selection, where $i \in \{0, 1, 2\}$}
\nomenclature{$C_{out}$}{Carry output from arithmetic operations}
\nomenclature{$C_{in}$}{Carry input to arithmetic operations}
\nomenclature{$\overline{A}$}{Bitwise complement (NOT) of operand A}
\nomenclature{$A + B$}{Arithmetic addition of operands A and B}
\nomenclature{$A - B$}{Arithmetic subtraction (A minus B)}
\nomenclature{$A \times B$}{Multiplication of operands A and B}
\nomenclature{$A \div B$}{Division of operand A by B (quotient)}
\nomenclature{$A \cdot B$}{Bitwise AND operation}
\nomenclature{$A + B$}{Bitwise OR operation (when in logic context)}
\nomenclature{$A \oplus B$}{Bitwise XOR (Exclusive OR) operation}
\nomenclature{$V_{CC}$}{Positive supply voltage (typically 5V for TTL logic)}
\nomenclature{$V_{DD}$}{Drain supply voltage for CMOS circuits}
\nomenclature{$t_{pd}$}{Propagation delay time through a logic gate or circuit}
